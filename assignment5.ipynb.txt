    "def ridge_gradient_descent(X, y, lr, lam, iterations):\n",
    "    m, n = X.shape\n",
    "    X = np.c_[np.ones(m), X]\n",
    "    beta = np.zeros((n + 1, 1))\n",
    "    for _ in range(iterations):\n",
    "        y_pred = X @ beta\n",
    "        y_pred = np.nan_to_num(y_pred)\n",
    "        error = y_pred - y\n",
    "        gradient = (X.T @ error + lam * beta) / m\n",
    "        gradient[0] = (X[:,0].T @ error) / m\n",
    "        beta -= lr * gradient\n",
    "    cost = (1/(2*m)) * np.sum((y_pred - y)**2) + (lam/(2*m))*np.sum(beta[1:]**2)\n",
    "    return beta, cost"
   ]
